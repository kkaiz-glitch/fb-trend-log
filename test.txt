import streamlit as st
import pandas as pd
from google.oauth2 import service_account
from gspread_pandas import Spread

st.set_page_config(page_title="F&B Trend Log", layout="wide")

st.title("📊 F&B Trend Log")
st.sidebar.header("설정")

# 💡 마케터님을 위한 팁: 
# 처음에는 드라이브 연결이 복잡하니, 
# 수집한 csv 파일을 GitHub에 같이 올리고 아래 코드로 읽어보세요.
try:
    df = pd.read_csv("insta_trend_master.csv")
    
    # 날짜 필터
    date_list = sorted(df['timestamp_kst'].unique(), reverse=True)
    selected_date = st.sidebar.selectbox("보고 싶은 날짜를 선택하세요", date_list)

    # 데이터 필터링
    filtered_df = df[df['timestamp_kst'] == selected_date]

    # 메인 대시보드 구성
    col1, col2 = st.columns(2)
    with col1:
        st.subheader(f"📅 {selected_date} 핵심 요약")
        # GPT 분석 결과나 주요 키워드를 여기에 뿌려줄 수 있습니다.
        st.write("해당 날짜의 수집된 캡션들을 기반으로 분석을 진행하세요.")

    with col2:
        st.subheader("📈 인게이지먼트 TOP 3")
        top_posts = filtered_df.nlargest(3, 'likesCount')
        st.dataframe(top_posts[['ownerUsername', 'likesCount', 'commentsCount']])

    st.divider()
    st.subheader("📱 전체 게시물 데이터")
    st.dataframe(filtered_df)

except Exception as e:
    st.error("데이터 파일(csv)을 GitHub 저장소에 먼저 올려주세요!")
